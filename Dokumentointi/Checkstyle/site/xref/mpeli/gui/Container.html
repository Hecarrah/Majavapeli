3   import java.awt.Color;
4   import java.awt.EventQueue;
5   import java.awt.event.ActionEvent;
6   import java.awt.event.ActionListener;
7   import javax.swing.JFrame;
8   import javax.swing.JPanel;
9   import javax.swing.SwingUtilities;
10  import javax.swing.Timer;
11   /**
12    * Container on graafisen käyttöliittymän ulkokuori. Luokka toimii pelin pääluokkana.
13    * @author Peter
14    */
15  public class Container extends JFrame implements ActionListener {
16      private static int sY = 0;
17      private static int sX = 0;
18      
19      private Timer timer;
20      public Container(){
21          init();
22      }
23      /**
24       * Init:issä luodaan containerin sisälle itse pelilauta joka huolehtii pelin kulusta, ja valmistellaan container pelin toimintaa varten asettamalla peliruudun koko ja lukitsemalla se.
25       * Init:issä aloitetaan myös timer joka pitää osaltaan huolen että ruutu pysyy päivittyneenä
26       */
27      private void init(){
28          Pelilauta p = new Pelilauta(); 
29          add(p);    
30          sY = 512;
31          sX = 512;
32          setSize(sX, sY);
33          setResizable(false);    
34          setTitle("Majavapeli"); 
35          setLocationRelativeTo(null);
36          setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
37          
38          setBackground(Color.black);
39          timer = new Timer(10,this);
40          timer.start();
41          
42      } /**
43       * Pelin main metodi joka aloittaa pelin toiminnan.
44       * Metodi myös asettaa itse containerin näkyväksi sen jälkeen kun kaikki siihe tarvittavat muut toimenpiteet ovat suoritettu.
45       * @param args 
46       */
47          public static void main(String[] args) {
48          SwingUtilities.invokeLater(new Runnable() {
49              @Override
50              public void run() {
51                  
52                  Container c = new Container();
53                  c.setVisible(true);
54              }
55          });
56      }
57          public static int getH(){
58              return sY;
59          }
60          public static int getW(){
61              return sX;
62          }
63  /**
64   * actionPerformed metodi huolehtii siitä että kun käyttäjä painaa mitä tahansa nappia, tai ylipäänsä tekee mitään, niin ruutu päivittää itsensä.
65   * @param e actionPerformed metodin vaatima parametri jota ei kuitenkaan käytetä tässä
66   */
67      @Override
68      public void actionPerformed(ActionEvent e) {
69          repaint();
70      }
71      public boolean getTimer(){
72          return timer.isRunning();
73      }
74  }
