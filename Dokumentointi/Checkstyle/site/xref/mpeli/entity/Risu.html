5   import java.util.Random;
6   /**
7    * risu luokka kuvaa risua joita pelaajan tulisi kerätä voittaakseen pelin.
8    * @author Peter
9    */
10  public class Risu extends Esine{
11      /**
12       * @param montako kuvaa sitä kuinka monta risua pelaajalla on hallussaan kullakin hetkellä.
13       */
14      int montako = 0;
15      private Image image;
16      
17      /** 
18       * konstruktorissa määritellään ensimmäisen risun sijainti, ja muutetaan esineen tyyppiä.
19       */
20      public Risu(){
21          init();
22          super.setPos(50, 50);
23          super.change("Risu");
24      }
25      /**
26       * kerätty metodissa nostetaan hallussa olevien risujen määrää yhdellä.
27       */
28      public void keratty(){
29          montako++;
30      }
31      /** 
32       * Randomize metodi antaa keräyksen jälkeen uudelle risulle uuden paikan, jotta pelaaja pääsee hakemaan uutta risua.
33       */
34      public boolean randomize(){
35          Random r = new Random();
36          boolean bo = r.nextBoolean();
37          double x = Math.random()*50;
38          double y = Math.random()*10;
39          if(bo){
40  
41          int a = (int)x*10;
42          int b = (int)y*10+350;
43          super.setPos(a, b);
44          return true;
45          }else{
46              
47          int a = (int)x*10;
48          int b = (int)y*10;
49          super.setPos(a, b);
50          return true;
51          }
52      }
53      /**
54       * patoon metodi toimii päinvastoin kun kerätty metodi, ja poistaa pelaajalta yhden risun.
55       */
56      public void patoon(){
57          montako--;
58      }
59      public int getMaara(){
60          return montako;
61      }
62      /**
63       * init metodissa annetaan risulle kuva jota käytetään graafisessa liittymassa
64       */
65      public void init(){
66          ImageIcon ii = new ImageIcon(Risu.class.getResource("/pun.jpg"));
67          image = ii.getImage();
68      }
69          public Image getImage(){
70          return image;
71      }
72  }
